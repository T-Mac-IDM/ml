gmm function implementation

function Px = gmm(X, K_or_centroids)  
% ============================================================  
%  
% PX = GMM(X, K_OR_CENTROIDS)  
%  
%  - X: N-by-D data.
%  - K_OR_CENTROIDS: either K indicating the number of  
%       components or a K-by-D matrix indicating the  
%       choosing of the initial K centroids.  
%  
%  - PX: N-by-K matrix indicating the probability of each  
%       component generating each point.  
% ============================================================  

  threshold = 1e-15;  
  [N, D] = size(X);  

  if isscalar(K_or_centroids) %判断是否为纯数字
    K = K_or_centroids;  
    Rn_index = randperm(N); %生成一串1~N的随机数串
    centroids = X(Rn_index(1:K), :); %读取上面生成的数串中的前K个数字，将X中的对应的点作为initial的初始化点
  else
    K = size(K_or_centroids, 1);   
    centroids = K_or_centroids;  
  end 

  %% pMiu-均值-K-D  pPi-权重-1-K  pSigma-协方差矩阵-D-D  
  [pMiu pPi pSigma] = init_params();  

  Lprev = -inf; %上一次聚类的误差

  %% EM Algorithm  
  while true  
    %% Estimation Step  
    Px = calc_prob();

    pGamma = Px .* repmat(pPi, N, 1); %乘上权重 still N-by-K 
    pGamma = pGamma ./ repmat(sum(pGamma, 2), 1, K); %分母 对每个样本的K个G产生的概率求和，作为每个样本的分母。

    Nk = sum(pGamma, 1); %每个高斯的概率和

    %% Maximization Step
    pPi = Nk / N;
    pMiu = diag(1./Nk) * pGamma' * X;
    for kk = 1:K   
        Xshift = X-repmat(pMiu(kk, :), N, 1);  
        pSigma(:, :, kk) = (Xshift' * ...  
            (diag(pGamma(:, kk)) * Xshift)) / Nk(kk);  
    end 

    % check for convergence  
    L = sum(log(Px*pPi'));  
    if L-Lprev < threshold  
        break;  
    end  
    Lprev = L;  
  end

function [pMiu pPi pSigma] = init_params()  
  pMiu = centroids; %用传入的centroids初始化
  pPi = zeros(1, K); %k类GMM所占权重（influence factor）  
  pSigma = zeros(D, D, K); %k类GMM的协方差矩阵，每个是D*D的  

  % N个样本到K个分类中心的距离矩阵，计算N*K的矩阵
  %(x-pMiu)^2 = (x(i,1) - pMiu(j,1))^2 + ... + (x(i,D) - pMiu(j,D))^2 <i=1:N j=1:K> 
  % =(x(i,1)^2 + ... + x(i,D)^2) + (pMiu(j,1)^2 + ... + pMiu(j,D)^2) - 2(x(i,1)*pMiu(j,1) + .. + x(i,D)*pMiu(j,D)))  

  distmat = repmat(sum(X.*X, 2), 1, K) + ... %x^2, N*1的矩阵replicateK列  
            repmat(sum(pMiu.*pMiu, 2)', N, 1) - ...%pMiu^2，1*K的矩阵replicateN行  
            2*X*pMiu';  
  [~, labels] = min(distmat, [], 2);  %Return the minimum from each row <2表示每row，[~, labels]其中labels为index> 

  for k=1:K  
    Xk = X(labels == k, :);  %这么写，相当于遍历labels，返回所有等于k的行
    pPi(k) = size(Xk, 1)/N;  
    pSigma(:, :, k) = cov(Xk);  %If each row of X and Y is an observation, and each column is a variable, then the (I, J)-th entry of `cov (X, Y)' is the covariance between the I-th variable in X and the J-th variable in Y.
    % If called with one argument, compute `cov (X, X)', the covariance between the columns of X.
  end  
end    

function Px = calc_prob()   
    %Gaussian posterior probability   
    %N(x|pMiu,pSigma) = 1/((2pi)^(D/2))*(1/(abs(sigma))^0.5)*exp(-1/2*(x-pMiu)'pSigma^(-1)*(x-pMiu))  
    Px = zeros(N, K);  
    for k = 1:K  
        Xshift = X-repmat(pMiu(k, :), N, 1); %X-pMiu  
        inv_pSigma = inv(pSigma(:, :, k));  
        tmp = sum((Xshift*inv_pSigma) .* Xshift, 2);  
        coef = (2*pi)^(-D/2) * sqrt(det(inv_pSigma));  
        Px(:, k) = coef * exp(-0.5*tmp);  
    end  
end  
end

x = mvnrnd([0,0], eye(2), 200);
y = mvnrnd([3,3], eye(2) + [1 1;1 1], 200);
z = mvnrnd([6,0], eye(2) + [2 2;2 2], 200);
d(1:200,1:2) = x;
d(201:400,1:2) = y;
d(401:600,1:2) = z;
plot(d(:,1), d(:,2), '*');

px = gmm(d,3);

[~ idx] = max(px, [], 2);

figure
plot (d (idx==1, 1), d (idx==1, 2), 'ro');
hold on
plot (d (idx==2, 1), d (idx==2, 2), 'bs');
plot (d (idx==3, 1), d (idx==3, 2), 'ys');
